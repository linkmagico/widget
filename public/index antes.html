Widget Embed Code Comercial

Cole este código em qualquer site para adicionar o chatbot autenticado: 


<script>
(function() {
    var config = {
        apiKey: 'SUA_API_KEY_AQUI', // OBRIGATÓRIO
        robotName: 'Seu Assistente',
        salesUrl: 'https://sua-pagina.com',
        instructions: 'Responda em até 3 frases',
        position: 'bottom-right',
        primaryColor: '#3b82f6'
    };
    var script = document.createElement('script');
    script.src = window.location.origin + '/widget.js';
    script.onload = function() {
        if (window.LinkMagicoWidget) {
            window.LinkMagicoWidget.init(config);
        }
    };
    document.head.appendChild(script);
})();
</script>
